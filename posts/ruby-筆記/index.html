number = Random.rand(14)+1
#while guess==number #迴圈
while true
  print("請猜一個1~15之間的數字")
  guess=gets.chomp.to_i
  if guess == number # '8' != 8
    puts '猜對了，答案是'+number.to_s
    break;
  else
    if guess > number
      puts '猜錯了! 太大了'
    else
      puts '猜錯了! 太小了'
    end
  end
end

# 1=>檢視成績 2 新增成績 3 刪除成績 4 結束成績
scores=[90,88,85]
puts '輸入 1 檢視成績、2 新增成績、3 刪除成績、4 結束程式'
while true
  print '>'
  command = gets.chomp.to_i

  if command == 1
    scores.each do |score|
      puts score
    end
  end
  if command == 2
    print '請輸入要新增的成績:'
    scores.push gets.chomp.to_i
  end
  if command == 3
    print '請輸入要刪除的成績位置:'
    scores.delete_at (gets.chomp.to_i-1)
  end
  if command == 4
    puts '程式結束!'
    break
  end
end
